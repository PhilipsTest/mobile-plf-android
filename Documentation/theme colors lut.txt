//These are the themes as they are currently defined in iOS

+ (PUITheme *)blueDark
{
    if (nil == PUIRealThemeDarkBlue) {
        PUIRealThemeDarkBlue = [[PUITheme alloc] initWithName:@"Dark Blue"
                                                    baseColor:[UIColor philipsBlueDark]
                                                  darkerColor:[UIColor philipsBlueVeryDark]
                                                 lighterColor:[UIColor philipsBlueBright]
                                               veryLightColor:[UIColor philipsBlueVeryLight]
                                                 baseGradient:[PUIGradient philipsBlueDark]
                                               darkerGradient:[PUIGradient philipsBlueVeryDark]
                                              lighterGradient:[PUIGradient philipsBlueBright]
                                          evenLighterGradient:[PUIGradient philipsBlueLight]
                                            veryLightGradient:[PUIGradient philipsBlueVeryLight]];
    }
    return PUIRealThemeDarkBlue;
}
